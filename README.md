   OPTIONAL: ENABLE THE SNAKE ANIMATION
  ---------------------------------------------------------
   1. In your profile repo, create this file:
        .github/workflows/snake.yml
   2. Paste the workflow below into it.
   3. Go to repo Settings → Actions → enable "Read and write".
   4. Run the action once manually (Actions tab → Run workflow).

  name: Generate Snake
  on:
    schedule:
      - cron: "0 0 * * *"
    workflow_dispatch:
  jobs:
    generate:
      runs-on: ubuntu-latest
      permissions:
        contents: write
      steps:
        - uses: Platane/snk@v3
          with:
            github_user_name: ${{ github.repository_owner }}
            outputs: |
              dist/github-contribution-grid-snake-dark.svg?palette=github-dark
        - uses: crazy-max/ghaction-github-pages@v4
          with:
            target_branch: output
            build_dir: dist
          env:
            GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
